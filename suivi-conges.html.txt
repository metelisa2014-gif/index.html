        function updateDate() {
            document.getElementById('currentDate').textContent = new Date().toLocaleDateString(lang === 'fr' ? 'fr-FR' : 'ar-TN', {weekday:'long', year:'numeric', month:'long', day:'numeric', hour:'2-digit', minute:'2-digit'});

        function toggleLang() {
            lang = lang === 'fr' ? 'ar' : 'fr';
            document.documentElement.dir = lang === 'ar' ? 'rtl' : 'ltr';
            document.getElementById('langBtn').textContent = lang === 'fr' ? 'ÇáÚŃČíÉ' : 'Français';
            updateDate();
        }

        function show(section) {
            document.querySelectorAll('.section').forEach(s => s.classList.add('hidden'));
            document.getElementById(section).classList.remove('hidden');
            document.querySelectorAll('.sidebar-item').forEach(s => s.classList.remove('active'));
            event.target.closest('.sidebar-item').classList.add('active');

            document.getElementById('pageTitle').textContent = titles[section];
            
            if(section === 'employees') renderEmployees();
            if(section === 'maladie') {renderMaladie(); updateSelects();}
            if(section === 'repos') {renderRepos(); updateSelects();}
            if(section === 'absences') {renderAbsences(); updateSelects();}
            if(section === 'alerts') renderAlerts();
            if(section === 'milk') renderMilk('all');
        }

        function logout() {
            document.getElementById('mainApp').classList.add('hidden');
            document.getElementById('loginPage').classList.remove('hidden');
        }

        // Excel Import
        function importExcel(e) {
            const file = e.target.files[0];
            const reader = new FileReader();
            reader.onload = (ev) => {
                const data = new Uint8Array(ev.target.result);
                const workbook = XLSX.read(data, {type:'array'});
                const json = XLSX.utils.sheet_to_json(workbook.Sheets[workbook.SheetNames[0]]);
                
                json.forEach(row => {
                    employees.push({
                        id: row.ID || row.id || Date.now(),

        // Repos
        function saveRepos(e) {
            e.preventDefault();
            const start = new Date(document.getElementById('rStart').value);
            const end = new Date(document.getElementById('rEnd').value);
            const days = Math.ceil((end-start)/(1000*60*60*24))+1;
            const used = repos.filter(r => r.empId == document.getElementById('rEmp').value && r.year == start.getFullYear()).reduce((s,r) => s+r.days, 0);
            
            repos.push({
                id: Date.now(),
                empId: document.getElementById('rEmp').value,
                start: document.getElementById('rStart').value,
                end: document.getElementById('rEnd').value,
                days: days,
                remaining: Math.max(0, 30-used-days),
                note: document.getElementById('rNote').value,
                month: start.getMonth()+1,
                year: start.getFullYear()
            });
            save();
            renderRepos();
            document.getElementById('reposForm').reset();
            checkAlerts();
        }

        // Alerts
        function checkAlerts() {
            const month = new Date().getMonth()+1;
            const year = new Date().getFullYear();
            let count = 0;
            
            employees.forEach(emp => {
                const m = maladies.filter(x => x.empId == emp.id && x.month == month && x.year == year).reduce((s,x) => s+x.days, 0);
                const r = repos.filter(x => x.empId == emp.id && x.month == month && x.year == year).reduce((s,x) => s+x.days, 0);
                const a = absences.filter(x => x.empId == emp.id && x.month == month && x.year == year).reduce((s,x) => s+(x.hours/8), 0);
                if(m+r+a >= 20) count++;
            });
            
            document.getElementById('alertBadge').textContent = count;
            document.getElementById('alertBadge').classList.toggle('hidden', count == 0);
            document.getElementById('notifBadge').textContent = count;
            document.getElementById('notifBadge').classList.toggle('hidden', count == 0);
            document.getElementById('statAlerts').textContent = count;
        }

        function renderAlerts() {
            const filterMonth = document.getElementById('alertMonth').value;
            const [y, m] = filterMonth ? filterMonth.split('-').map(Number) : [new Date().getFullYear(), new Date().getMonth()+1];
            
            document.getElementById('alertTable').innerHTML = employees.map(emp => {
                const mal = maladies.filter(x => x.empId == emp.id && x.month == m && x.year == y).reduce((s,x) => s+x.days, 0);
                const rep = repos.filter(x => x.empId == emp.id && x.month == m && x.year == y).reduce((s,x) => s+x.days, 0);
                const abs = absences.filter(x => x.empId == emp.id && x.month == m && x.year == y).reduce((s,x) => s+(x.hours/8), 0);
                const total = mal + rep + abs;
                
                if(total >= 20) {
                    return `<tr class="alert-row"><td class="p-3 font-bold">${emp.nom} ${emp.prenom}</td><td class="p-3 text-red-600">${mal}</td><td class="p-3 text-green-600">${rep}</td><td class="p-3 text-orange-600">${abs.toFixed(1)}</td><td class="p-3 font-bold text-xl">${total.toFixed(1)}</td><td class="p-3"><span class="bg-red-500 text-white px-2 py-1 rounded text-sm">ALERTE</span></td></tr>`;
                }
                return '';
            }).join('');
        }

        // PDF Export
        function exportPDF(type) {
            const {jsPDF} = window.jspdf;
            const doc = new jsPDF();
            
            doc.setFillColor(30,58,138);
            doc.rect(0,0,210,40,'F');
            doc.setTextColor(255,255,255);
            doc.setFontSize(18);
            doc.text('Direction Régionale de la Justice - Kasserine', 105, 20, {align:'center'});
            doc.setFontSize(14);
            
            const titles = {general:'Rapport Général', maladie:'Rapport Maladie', repos:'Rapport Repos', alerts:'Rapport Alertes', milk:'Rapport Prime de Lait'};
            doc.text(titles[type], 105, 30, {align:'center'});
            
            let y = 50;
            const m = new Date().getMonth()+1;
            const yr = new Date().getFullYear();
            doc.setTextColor(0,0,0);
            doc.text(`Période: ${m}/${yr}`, 20, y);
            
            if(type == 'general') {
                const data = employees.map(emp => {
                    const mal = maladies.filter(x => x.empId == emp.id && x.month == m && x.year == yr).reduce((s,x) => s+x.days, 0);
                    const rep = repos.filter(x => x.empId == emp.id && x.month == m && x.year == yr).reduce((s,x) => s+x.days, 0);
                    const abs = absences.filter(x => x.empId == emp.id && x.month == m && x.year == yr).reduce((s,x) => s+(x.hours/8), 0);
                    return [emp.id, emp.nom+' '+emp.prenom, emp.grade, mal, rep, abs.toFixed(1), (mal+rep+abs).toFixed(1)];
                });
                doc.autoTable({startY:y+10, head:[['ID','Nom','Grade','Maladie','Repos','Absences','Total']], body:data, theme:'grid', headStyles:{fillColor:[30,58,138]}});
            }
            
            doc.save(`rapport_${type}_${m}_${yr}.pdf`);
        }
